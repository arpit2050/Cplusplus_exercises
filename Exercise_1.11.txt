#include <iostream>
int main()
{
int a,b;
std::cin>>a>>b;

if(a>b)
{
    while(a>=b)
    {
        std::cout<<a<<std::endl;
        a--;
    }
}
else if(a<b)
{
      while(a<=b)
    {
        std::cout<<a<<std::endl;
        a++;
    }
}
else
{
    std::cout<<a<<" and "<<b<<" are same ";
}

return 0;
}
